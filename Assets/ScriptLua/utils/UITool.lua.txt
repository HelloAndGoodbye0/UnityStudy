UITool = {}
canvas = nil
UITool.addBtnCall = function (gameObj,call)
    if gameObj:GetComponent("Button") then
        gameObj:GetComponent("Button").onClick:AddListener(function()
            if call then
                call()
            end
        end)
    end
end

UITool.showAlert = function (title,content,call)
    local obj = GameObject.Instantiate(Resources.Load("prefabs/alert"))
    
    local pos = obj.transform.position
    obj.transform:SetParent(canvas)
    print(pos.x,pos.y)
    
end

UITool.setCanvas = function (gameObj)
    canvas = gameObj
end
