require("utils.global")
require("utils.UITool")

function awake()
    print("loginScene awake")
end


function onStart()
    print("loginScene Start")
    UITool.addBtnCall(self.transform:Find("btn_login"),function()
        -- SceneManager.LoadScene("main");
        local textNode = self.transform:Find("btn_login/Text")
        local active = textNode.gameObject.activeSelf
        textNode.gameObject:SetActive(not active )
    end)

    text:GetComponent(typeof(CS.UnityEngine.UI.Text)).text = "Hello World"
end

function onEnable()
    print("loginScene onEnable")
end

    print("loginScene onDisable")    
end

end
