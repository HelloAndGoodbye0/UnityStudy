require("utils.global")
require("utils.UITool")
function awake()
    print("loginScene awake")
    -- print(CS.UnityEngine.Application.persistentDataPath)
    -- File.WriteAllText(CS.UnityEngine.Application.persistentDataPath.."/test.txt","test")
    UITool.setCanvas(self.transform.parent)
end


function onStart()
    print("loginScene onStart",self)
    UITool.addBtnCall(self.transform.parent:Find("btn_login"),function()
        -- SceneManager.LoadScene("main");
        local textNode = self.transform.parent:Find("btn_login/Text")
        local active = textNode.gameObject.activeSelf
        textNode.gameObject:SetActive(not active )

        UITool.showAlert()
    end)

    text:GetComponent(typeof(CS.UnityEngine.UI.Text)).text = "Hello World !"
end

function onEnable()
    print("loginScene onEnable")
end

function onDisable()
    print("loginScene onDisable")    
end



function onDestroy()
    print("loginScene onDestroy")
end
